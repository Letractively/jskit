区域代码： 
var jasPop = new JskitAreaSelector("jasPop");
function doAreaSelect(){
	jasPop.open({
		data:JskitAreaSelectorData,//必须，区域数据
		textFieldId : "txtArea",
		valueFieldId : "hidArea",
		startLevel:2,//可选，开始的级别，缺省为2;1:国家,2:省
		endLevel:4,//可选，结束的级别，缺省为4;3:市,4:区县.同开始级别，如果结束在3级，则只选择到市，不再选择区县
		separate:",",//可选，显示文本内容的分隔符，缺省为逗号
		useKeyAsValue:false, //true:使用区域的代码,false:使用区域的名字		
		onSelected : function(key,txtArray){
			var _prov = txtArray[0];
			var _city = txtArray[1];
			alert("省:"+_prov+"\n市:"+_city);
		}
	});
}
级联方式 
var jasSelector = new JskitAreaSelector("jasSelector");
function PageOnLoad(){
	jasSelector.select({
		data:JskitAreaSelectorData,//必须，区域数据
		selectFieldIdList:["ddlProvince","ddlCity","ddlDistrict"],
		values:["","",""],//设置缺省值，在编辑时在这里为控件赋值
		empty : {key:"",value:"--请选择--"},//如果不设置，则第一个下拉列表自动选中第一项
		startLevel:2,//可选，开始的级别，缺省为2;1:国家,2:省
		separate:",",//可选，显示文本内容的分隔符，缺省为逗号
		useKeyAsValue:false, //true:使用区域的代码,false:使用区域的名字
		onSelected : function(key,txtArray){
			//设置在选择完成之后调用的方法，该方法被传递两个参数：
			//key: 区域的Key
			//txtArray: 选择区域的各级名称，根据startLevel来决定。
			var _prov = txtArray[0];
			var _city = txtArray[1];
			var _dist = txtArray[2];
			alert("省:"+_prov+"\n市:"+_city+"\n区县:"+_dist);
		}
	});
}
